def interpolate(x_values, y_values, x):
    def basis(j):
        p = [(x - x_values[m])/(x_values[j] - x_values[m]) for m in range(k) if m != j]
        return reduce(lambda x, y : x * y, p)
    k = len(x_values)
    return sum(basis(j) * y_values[j] for j in range(k))
